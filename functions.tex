\NewDocumentCommand{\nameHeader}{m}{
  \Huge{
    \textbf{%
      \textsc{#1}
    }
  }
}

\NewDocumentCommand{\link}{m}{
  \small{%
    \underline{#1}
  }%
}

\NewDocumentCommand{\underlinedSection}{m}{
  \section*{#1}
  \vspace{-1em}
  \hrule
  \vspace{0.8em}
}
